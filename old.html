Optional Argument

  var hello = "Hello";

Arguments Object

var sum = function(){
  var l = arguments.length;
  var sum = 0;
  var i;
  for (i=0; i<l; i += 1){
    sum += arguments[i];
  }
  return sum;
}

alert(sum(1,2,3,4,5));

For...In

function Person(name, age){
  this.name = name;
  this.age = age;
}

Person.prototype.species = "human";

Person.prototype.displayOwnProperties = function(){
  var dispArr;
  for(property in this){
    if (this.hasOwnProperty(property)){
      dispArr = [property, ": ", this[property], "<br />"];
      document.write( dispArr.join("") );
    }
  }
}

var bob = new Person("Bob", 29);
bob.displayOwnProperties();
Demo

Getter Setter

function GetterSetter(){
  var x = 0;
  this.get = function(){ return x; }
  this.set = function(setX){
    if (typeof(setX) === "number" && setX >= 0){
      x = setX;
    }
  }
}

var gs = new GetterSetter();
gs.set(10);
document.write(gs.get() + ", ");
gs.set(5);
document.write(gs.get() + ", ");
gs.set(-1);
document.write(gs.get());
Demo
