Optional Argument

var SayHello = function(name){
  var hello = "Hello";
  if ( name !== undefined ){
    hello += ", my name is " + name;
  }
  alert(hello);
}

SayHello(); // Hello
SayHello("Joe"); //Hello, my name is Joe
Demo

Arguments Object

var sum = function(){
  var l = arguments.length;
  var sum = 0;
  var i;
  for (i=0; i<l; i += 1){
    sum += arguments[i];
  }
  return sum;
}

alert(sum(1,2,3,4,5));
Demo

Scope

var n = 5;
(function(){
  for(var i=0; i<10; i += 1){
    var j = i*n;

  document.write(i +", " + j);

Scope - This is Better

var n = 5;
(function(){
  var i,j;
  var local_n = n;
  for(i=0; i<10; i += 1){
    j = i*local_n;
  }
}());
  • Variable declaration at top of scope
  • Scope can be expensive

For...In

function Person(name, age){
  this.name = name;
  this.age = age;
}

Person.prototype.species = "human";

Person.prototype.displayOwnProperties = function(){
  var dispArr;
  for(property in this){
    if (this.hasOwnProperty(property)){
      dispArr = [property, ": ", this[property], "<br />"];
      document.write( dispArr.join("") );
    }
  }
}

var bob = new Person("Bob", 29);
bob.displayOwnProperties();
Demo

Basic Closure

var bar = new Foo(5);

Demo

Getter Setter

function GetterSetter(){
  var x = 0;
  this.get = function(){ return x; }
  this.set = function(setX){
    if (typeof(setX) === "number" && setX >= 0){
      x = setX;
    }
  }
}

var gs = new GetterSetter();
gs.set(10);
document.write(gs.get() + ", ");
gs.set(5);
document.write(gs.get() + ", ");
gs.set(-1);
document.write(gs.get());
Demo

Object Builder

var Person = function(name){
  var self = {
    name: function(){
      return name;
    }
  };
  return self;
}

var bob = Person("Bob");
alert(bob.name());

  • Widely Supported (IE8.0+, FF3.5+, CH4.0+) support

getElementById

var hello = document.getElementById("hello");
  • Faster than QuerySelector and jQuery
  • Basically supported everywhere

getElementsByClassName

var theList = document.getElementById("theList");
var liNodeList = document.getElementsByClassName("person");
alert(liNodeList.length);

var demoButton = document.getElementById('demoButton');
demoButton.onclick = liveNodeListDemoFunction;

var addLi = "<li class='person'>Some other person</li>";
function liveNodeListDemoFunction(){
  theList.innerHTML += addLi;
  alert(liNodeList.length);
}
Demo
